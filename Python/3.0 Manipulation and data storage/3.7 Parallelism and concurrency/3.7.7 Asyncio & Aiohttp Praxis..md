# Asyncio

```python
import asyncio

async def think_deeply(thought, seconds_needed):
	""" 
	Simulates a task that requires waiting time. 
	
	Mental model: Each thought is a coroutine that surrenders 
	control while it processes. 
	"""
	print(f"Begining thought: '{thought}'")
	
	#Surrender control to loop event for this duration
	await asyncio.sleep(seconds_needed)
	
	print(f"Thought complete: '{thought}'") 
	return f"Insight from: {thought}"

async def think_currently():
	""" 
	Orchestrates multiple thoughts simultaneously. 
	
	Mental model: A philosopher processing multiple questions 
	at once — not sequentially, concurrently. 
	"""
	
	#Schedule all three thoughts simultaneously
	
	all_insights = await asyncio.gather(
		think_deeply("What is consciousness?", seconds_needed=3),
		think_deeply("What is time?", seconds_needed=2),
		think_deeply("What is reality?", seconds_needed=1),
	)
	return all_insights

#The entry point — this STARTS the event loop 
insights = asyncio.run(think_concurrently())

```

# The core asyncio toolkit

```python
import asyncio


# TOOL 1: asyncio.run()
# Starts the event loop and runs until the coroutine completes
# This is always your entry point
asyncio.run(your_main_coroutine())


# TOOL 2: asyncio.gather()
# Schedules multiple coroutines simultaneously
# Waits for ALL of them to complete
# Returns results in original order
results = await asyncio.gather(
    coroutine_one(),
    coroutine_two(),
    coroutine_three()
)


# TOOL 3: asyncio.sleep()
# The async version of time.sleep()
# SURRENDERS control during the wait
# Never use time.sleep() inside async code — it BLOCKS everything
await asyncio.sleep(seconds=2)


# TOOL 4: asyncio.create_task()
# Schedules a coroutine to run "in the background"
# Doesn't wait immediately — continues executing current code
background_task = asyncio.create_task(
    run_something_in_background()
)
# Continue doing other things here...
result = await background_task  # Collect result when needed
```



# Aiohttp

`aiohttp` is `requests` **rebuilt from the ground up for async.**

```python
import asyncio
import aiohttp


async def fetch_single_url(
    client_session,
    target_url
):
    """
    Fetches a single URL asynchronously.
    
    Mental model: We send the request, then AWAIT the response.
    During that await, the event loop serves other coroutines.
    We're not blocking. We're waiting intelligently.
    """
    async with client_session.get(target_url) as server_response:
        response_data = await server_response.json()
        return response_data


async def fetch_multiple_urls_concurrently(list_of_urls):
    """
    Fetches multiple URLs simultaneously using one session.
    
    Mental model: One client, many simultaneous conversations.
    Like one diplomat conducting multiple negotiations at once.
    """
    # ClientSession is the async HTTP client
    # async with ensures proper cleanup when done
    async with aiohttp.ClientSession() as client_session:
        
        # Schedule ALL requests simultaneously
        all_responses = await asyncio.gather(*[
            fetch_single_url(client_session, url)
            for url in list_of_urls
        ])
        
        return all_responses


# Run it
list_of_api_urls = [
    "https://api.example.com/user/1",
    "https://api.example.com/user/2",
    "https://api.example.com/user/3",
    "https://api.example.com/user/4",
    "https://api.example.com/user/5",
]

results = asyncio.run(
    fetch_multiple_urls_concurrently(list_of_api_urls)
)
```

# The client session - why it exists

```python
# You might wonder: why async with aiohttp.ClientSession()?
# Why not just call aiohttp.get() directly?

# ClientSession maintains:
# - A connection pool (reuses existing connections)
# - Cookie storage across requests
# - Default headers for all requests
# - Authentication credentials

# Creating a new TCP connection costs ~100ms
# Reusing an existing connection costs ~1ms
# ClientSession manages this automatically

# WRONG — creates new connection every request
async def inefficient_approach(url):
    async with aiohttp.ClientSession() as session:  # New session every time
        async with session.get(url) as response:
            return await response.json()

# RIGHT — one session, many requests
async def efficient_approach(list_of_urls):
    async with aiohttp.ClientSession() as session:  # One session total
        results = await asyncio.gather(*[
            fetch_with_session(session, url)
            for url in list_of_urls
        ])
    return results
```


# The Complete Picture — Both Libraries Together

```python
import asyncio
import aiohttp

async def fetch_ai_model_predictions(list_of_inputs):
	
	""" Sends multiple inputs to an AI API concurrently. 
	
	Mental model: Imagine sending 100 questions to an AI 
	simultaneously instead of one by one. This is how production 
	ML systems work at scale. 
	
	Args: list_of_inputs: Data points needing AI inference 
	
	Returns: List of AI predictions in original order 
	"""
	
	async with aiohttp.ClientSession() as client_session:
		
		prediction_tasks = [
			request_single_prediction(client_session, input_data)
			for input_data in list_of_inputs
		]
		
		# asyncio.gather runs ALL predictions concurrently 
		all_predictions = await asyncio.gather(*prediction_tasks) 
		
		return all_predictions


async def request_single_prediction(client_session, input_data):
	"""
	Request one prediction from the AI API
	
	Mental Model: One conversation with the AI.
	Await responses withoout blocking other conversations
	"""
	
	request_payload = { 
	"model": "your_ml_model", 
	"input": input_data 
	} 
	
	async with client_session.post( 
		url="https://your-ai-api.com/predict", 
		json=request_payload
	) as api_response: 
	
		prediction_result = await api_response.json() 
		return prediction_result 
	

#Entry Point
asyncio.run(
	fetch_ai_model_predictions(
		list_of_inputs = ["data_1", "data_2", "data_3"]
	)
)
```
